27 void adc0_config()
28 {
29  // Uses the same settings as recommended for calibration (AVG=32, ADC_CLK<4MHz)
30  // ADC_CLK = (BUSCLK/2) / 8 = (60/2)/8=3.75MHz. Max ADC_CLK is 24MHz however so a conversion just takes a little more time. Maybe more accurate though.
31  ADC0->CFG1 = ADC_CFG1_ADIV(3) | ADC_CFG1_ADLSMP(0) | ADC_CFG1_MODE(3) | ADC_CFG1_ADICLK(2); // ADIV=1:8, ADSLMP=short, MODE=16bit, ADICLK=BUSCLK/2
32  ADC0->CFG2 = ADC_CFG2_MUXSEL(0) | ADC_CFG2_ADACKEN(1) | ADC_CFG2_ADHSC(0) | ADC_CFG2_ADLSTS(0); // MUX=ch_A, ADACKEN=en, ADHSC=normal, ADLSTS=default
33  ADC0->SC2 = ADC_SC2_ADTRG(0) | ADC_SC2_ACFE(0) | ADC_SC2_ACFGT(0) | ADC_SC2_ACREN(0) | ADC_SC2_DMAEN(0) | ADC_SC2_REFSEL(0); // ADTRG=soft_trig, ACFE=comp_dis, ACFGT=x, ACREN=x, DMAEN=dis, REFSEL=default
34  ADC0->SC3 = ADC_SC3_CAL(0) | ADC_SC3_ADCO(0) | ADC_SC3_AVGE(1) | ADC_SC3_AVGS(3); // CAL=off, ADCO=single, AVGE=en, AVGS=32
35  ADC0->SC1[0] = 16; // Select ADC0_SE16 input, the only channel from ADC0 that's used on the Freedom Controller
36 }

125 uint16_t adc_read(uint8_t freedom_ain_channel)
126 {
127  uint16_t adc_res;
128  uint8_t ch_map[4] = {16, 16, 18, 23}; // Maps freedom_ain_channel to actual ADC0/1 channels
129 
130  //
131  freedom_ain_channel &= 0x03;
132 
133  if (freedom_ain_channel == 0) {
134  // ADC0: ADC0_SE16
135  // In software trigger mode, a write to SC1A starts a conversion.
136  ADC0->SC1[0] = ADC_SC1_ADCH( ch_map[freedom_ain_channel] );
137  // Wait for conversion_complete flag
138  while ( !(ADC0->SC1[0] & ADC_SC1_COCO_MASK) );
139  adc_res = ADC0->R[0];
140  } else {
141  // ADC1: ADC1_SE16 / ADC1_SE18 / ADC1_SE23
142  ADC1->SC1[0] = ADC_SC1_ADCH( ch_map[freedom_ain_channel] );
143  while ( !(ADC1->SC1[0] & ADC_SC1_COCO_MASK) );
144  adc_res = ADC1->R[0];
145  }
146  return adc_res;
147 }
