186 void freedom_mcp23017_init() {
187  mcp.reset();
188  mcp.config(
189  (MCP_GPB_DIR << 8) | MCP_GPA_DIR, // Set B to input except for B4; Set input on A0,A1,A6,A7 and output on A2,A3,A4,A5.
190  (MCP_GPB_PULL << 8) | MCP_GPA_PULL, // Set B to pull-up except for B4; Set pull-up for A0,A1,A6,A7.
191  (MCP_GPB_POL << 8) | MCP_GPA_POL); // Flip polarity for B0-B3; Flip polarity for A0,A1,A6,A7
192 
193  // Zero all pins
194  mcp.digitalWordWrite((MCP_GPB_VAL << 8) | MCP_GPA_VAL);
195 
196  // Set interrupts.
197  mcp.writeRegister(GPINTEN, (uint16_t)((MCP_GPB_INT << 8) | MCP_GPA_INT)); // Set interrupt pins
198  mcp.readRegister(GPINTEN);
199 }
